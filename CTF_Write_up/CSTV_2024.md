def Fermat_attack(n: int) -> tuple:
    if n % 2 == 0:
        return 2, n//2
    
    a = iroot(n, 2)[0] + 1
    b = iroot(a ** 2 - n, 2)[0]

    while (a + b) * (a - b) != n:
        a += 1
        b = iroot(a ** 2 - n, 2)[0]
    
    return a + b, a - b
